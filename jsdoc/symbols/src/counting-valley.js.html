1 /**
  2  * Counts number of valleys treversed by a hiker. 
  3  * Every step is denoted by: "D" for uphill climb and "D" for downhill steps.
  4  * <p>
  5  *  <strong>Note:</strong><br/>
  6  *  <ul>
  7  *      <li>A mountain is a sequence of consecutive steps above sea level, 
  8  *          starting with a step up from sea level and ending with a step down to sea level.</li>
  9  *      <li>A valley is a sequence of consecutive steps below sea level, 
 10  *          starting with a step down from sea level and ending with a step up to sea level.</li>
 11  *  </ul>
 12  * </p>
 13  * Full problem statement can be found at <a href="http://hr.gs/3rtx #programming">Hackerrank</a>
 14  * 
 15  * @param {int} n the number of steps
 16  * @param {String} s a string describing his path i.e. DDUUD
 17  */
 18 function countingValleys(n, s) {
 19     let isValley = false;
 20     let valley = 0;
 21     let latitude = 0;
 22     for(let i = 0; i < n; i++){
 23         if(s.charAt(i) == 'D'){
 24             latitude--;
 25         } else {
 26             latitude++;
 27         }
 28         if(isValley == false && latitude < 0){
 29             isValley = true;
 30         }
 31         if(isValley && latitude == 0){
 32             valley++;
 33             isValley = false;
 34         }
 35     }
 36     return valley;
 37 }
 38 
 39 module.exports = {
 40     countValleys: function(n, s){
 41         return countingValleys(n, s);
 42     }
 43 }
